    3. Per D-14: Add a useEffect that runs on mount (after tracks are initialized) to load persisted tracks from IndexedDB:
       ```typescript
       useEffect(() => {
         if (!projectId || tracks.length === 0) return;
         let cancelled = false;
         loadAllMusicTracks(projectId, gaits).then((persisted) => {
           if (cancelled) return;
           setTracks((prev) =>
             prev.map((t) => {
               const saved = persisted.get(t.gait);
               if (!saved) return t;
               const url = URL.createObjectURL(saved.blob);
               return {
                 ...t,
                 audioBlob: saved.blob,
                 audioUrl: url,
                 prompt: saved.prompt || t.prompt,
                 lyrics: saved.lyrics || t.lyrics,
                 promptEdited: !!saved.prompt,
                 lyricsEdited: !!saved.lyrics,
               };
             })
           );
         });
         return () => { cancelled = true; };
       }, [projectId, tracks.length === 0]);  // Only run once tracks are initialized
       ```
       NOTE: The dependency should be `tracks.length === 0` as a boolean (or use a ref to track if initial load happened). Use a `useRef(false)` called `loadedRef` to ensure this only runs once:
       ```typescript
       const loadedRef = useRef(false);
       useEffect(() => {
         if (!projectId || tracks.length === 0 || loadedRef.current) return;
         loadedRef.current = true;
         // ... load logic
       }, [projectId, tracks]);
       ```
